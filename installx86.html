MiraDB Linux Installation Guide


root@localhost:~# mkdir /home/miradb
root@localhost:~# cd /home/miradb
root@localhost:~# wget https://github.com/Nodeclient/Mira-DB/releases/download/linux/miradb-0.24-lin-x86.tar.gz
root@localhost:~# sudo tar xvzf miradb-0.24-lin-x86.tar.gz
root@localhost:~# sudo chmod 777 miradb
root@localhost:~# sudo rm miradb-0.24-lin-x86.tar.gz
root@localhost:~# sudo ./miradb
